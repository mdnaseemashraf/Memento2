def test  
    method = ENV["TEST_METHOD"]
    if method == "INSTRUMENTED"
        puts "UI Testing"

        # Build
        sh "./gradlew assembleGlobalDebug assembleGlobalDebugAndroidTest -PdisablePreDex"

        # Send to firebase
        sh " bash -x send_instrumented_test_to_firebase.sh"
    else 
        puts "unsupported TEST_METHOD arg"
    end
end
